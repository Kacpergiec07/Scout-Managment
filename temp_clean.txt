interface Standing {
  teamId: string;
  teamName: string;
  teamLogo: string;

  drawn: string;
  lost: string;

interface LeagueDetailsClientProps {

  initialLeagueInfo: any;
}

// Helper function to safely get match statistics
const getMatchStats = (match: any, side: 'home' | 'away', statKey: string, fallback: string = "0") => {
  const participant = match[`${side}Participant`];
  // Check teamStats first (correct API structure)
  if (participant?.teamStats?.[statKey] !== undefined && participant?.teamStats?.[statKey] !== null) {
    return participant.teamStats[statKey];
  }
  // Try direct properties as fallback
  const directKey = `${side}${statKey.charAt(0).toUpperCase() + statKey.slice(1)}`;
  return match[directKey] || fallback;
};
